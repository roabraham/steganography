207  public function encrypt($x) {
208  $t = ''; // 16-byte block to hold the temporary input of the cipher
209  $y = ''; // returned cipher text;
210  $y_block = $this->iv; // 16-byte block to hold the temporary output of the cipher
211  $xsize = strlen($x);
212 
213  switch ($this->mode) {
214  case 'ECB':
215  {
216  // put a 16-byte block into t, ecnrypt it and add it to the result
217  for ($i = 0; $i < $xsize; $i += 16) {
218  for ($j = 0; $j < 16; $j++) {
219  if (($i + $j) < $xsize) {
220  $t[$j] = $x[$i + $j];
221  } else {
222  $t[$j] = chr(0);
223  }
224  }
225 
226  $y_block = $this->encryptBlock($t);
227  $y .= $y_block;
228  }
229  break;
230  }
231  case 'CBC':
232  {
233  // put a 16-byte block into t, ecnrypt it and add it to the result
234  for ($i = 0; $i < $xsize; $i += 16) {
235  for ($j = 0; $j < 16; $j++) {
236  // XOR this block of plaintext with the initialization vector
237  $t[$j] = chr(ord(($i + $j) < $xsize ? $x[$i + $j] : chr(0)) ^ ord($y_block[$j]));
238  }
239

241  $y .= $y_block;
242  }
243  break;
244  }

361  public function encryptBlock($x) {
362  $y = ''; // 16-byte string
363  // place input x into the initial state matrix in column order
364  for ($i = 0; $i < 4 * self::$Nb; $i++) {
365  // we want integerger division for the second index
366  $this->s[$i % 4][($i - $i % self::$Nb) / self::$Nb] = ord($x[$i]);
367  }
368 
369  // add round key
370  $this->addRoundKey(0);
371 
372  for ($i = 1; $i < $this->Nr; $i++) {
373  // substitute bytes
374  $this->subBytes();
375 
376  // shift rows
377  $this->shiftRows();
378 
379  // mix columns
380  $this->mixColumns();
381 
382  // add round key
383  $this->addRoundKey($i);
